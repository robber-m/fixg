26fn sanitize(s: &str) -> String {
27    s.chars()
28        .map(|c| if c.is_ascii_alphanumeric() { c } else { '_' })
29        .collect()
30}

94#[cfg(feature = "aeron-ffi")]
95impl AeronMessageStore {
96    pub fn new_with_params(channel: &str, stream_id: i32) -> std::io::Result<Self> {
97        let client = AeronClient::connect()?;
98        let data_pub = Publication::add(&client, channel, stream_id)?;
99        let index_stream_id = stream_id + 1;
100        let index_pub = Publication::add(&client, channel, index_stream_id)?;
101        let index_sub = Subscription::add(&client, channel, index_stream_id)?;
102        Ok(Self { client, data_pub, index_pub, index_sub, channel: channel.to_string(), data_stream_id: stream_id, index_stream_id })
103    }
104    pub fn new() -> Self { panic!("use new_with_params") }
105
106    fn encode_index_frame(seq: u32, payload: &[u8]) -> Vec<u8> {
107        let mut v = Vec::with_capacity(4 + payload.len());
108        v.extend_from_slice(&seq.to_be_bytes());
109        v.extend_from_slice(payload);
110        v
111    }
112}

115#[async_trait]
116impl MessageStore for AeronMessageStore {
117    async fn append(&self, record: StoredMessageRecord) -> std::io::Result<()> {
118        // Fallback to bytes path if possible
119        if let Ok(bytes) = base64::decode(&record.payload_b64) {
120            self.append_bytes(&record.session, record.direction, record.seq, record.ts_millis, &bytes).await
121        } else {
122            Ok(())
123        }
124    }
125
126    async fn append_bytes(&self, _session: &SessionKey, direction: Direction, seq: Option<u32>, _ts_millis: u64, payload: &[u8]) -> std::io::Result<()> {
127        if direction == Direction::Outbound {
128            if let Some(s) = seq {
129                let _ = self.data_pub.offer_retry(payload, 10, 100, 1)?;
130                let idx = Self::encode_index_frame(s, payload);
131                let _ = self.index_pub.offer_retry(&idx, 10, 100, 1)?;
132            }
133        }
134        Ok(())
135    }
136
137    async fn load_outbound_range(&self, _session: &SessionKey, begin_seq: u32, end_seq: u32) -> std::io::Result<Vec<Bytes>> {
138        let frags = self.index_sub.poll_collect(100, 25);
139        let mut out: Vec<(u32, Bytes)> = Vec::new();
140        for f in frags.into_iter() {
141            if f.len() < 4 { continue; }
142            let seq = u32::from_be_bytes([f[0], f[1], f[2], f[3]]);
143            if seq >= begin_seq && seq <= end_seq {
144                out.push((seq, Bytes::from(f[4..].to_vec())));
145            }
146        }
147        out.sort_by_key(|(s, _)| *s);
148        Ok(out.into_iter().map(|(_, b)| b).collect())
149    }
150
151    async fn last_outbound_seq(&self, _session: &SessionKey) -> std::io::Result<Option<u32>> { Ok(None) }
152}

154pub fn make_store(backend: &StorageBackend) -> Arc<dyn MessageStore> {
155    match backend {
156        StorageBackend::File { base_dir } => Arc::new(FileMessageStore::new(base_dir.clone())),
157        StorageBackend::Aeron { archive_channel, stream_id } => {
158            #[cfg(feature = "aeron-ffi")] {
159                Arc::new(AeronMessageStore::new_with_params(archive_channel, *stream_id).expect("AeronMessageStore init"))
160            }
161            #[cfg(not(feature = "aeron-ffi"))] {
162                Arc::new(FileMessageStore::new("data/journal"))
163            }
164        }
165    }
166}

169pub struct FileMessageStore {
170    tx: mpsc::Sender<StoredMessageRecord>,
171    cfg: StorageConfig,
172}
173
174impl FileMessageStore {
175    pub fn new(base_dir: impl Into<PathBuf>) -> Self {
176        Self::new_with_config(StorageConfig { base_dir: base_dir.into(), ..StorageConfig::default() })
177    }
178
179    pub fn new_with_config(cfg: StorageConfig) -> Self {
180        let (tx, mut rx) = mpsc::channel::<StoredMessageRecord>(cfg.channel_capacity);
181        let cfg_clone = cfg.clone();
182        tokio::spawn(async move {
183            let _ = fs::create_dir_all(&cfg_clone.base_dir).await;
184            let mut queue: VecDeque<StoredMessageRecord> = VecDeque::with_capacity(cfg_clone.batch_max);
185            let mut ticker = time::interval(Duration::from_millis(cfg_clone.flush_interval_ms));
186            let mut last_sync: Instant = Instant::now();
187
188            loop {
189                tokio::select! {
190                    maybe = rx.recv() => {
191                        match maybe {
192                            Some(rec) => { queue.push_back(rec); },
193                            None => { flush_batch(&cfg_clone, &mut queue, &mut last_sync).await.ok(); break; }
194                        }
195                        if queue.len() >= cfg_clone.batch_max { let _ = flush_batch(&cfg_clone, &mut queue, &mut last_sync).await; }
196                    }
197                    _ = ticker.tick() => {
198                        if !queue.is_empty() { let _ = flush_batch(&cfg_clone, &mut queue, &mut last_sync).await; }
199                    }
200                }
201            }
202        });
203        Self { tx, cfg }
204    }
205}
206
207async fn flush_batch(cfg: &StorageConfig, queue: &mut VecDeque<StoredMessageRecord>, last_sync: &mut Instant) -> std::io::Result<()> {
208    while let Some(rec) = queue.pop_front() {
209        let stem = rec.session.file_stem();
210        let data_path = cfg.base_dir.join(format!("{}.jsonl", stem));
211        let idx_path = cfg.base_dir.join(format!("{}.idx", stem));
212
213        // Compute current offset before writing
214        let offset = match metadata(&data_path).await { Ok(m) => m.len(), Err(_) => 0 };
215
216        let mut f = OpenOptions::new().create(true).append(true).open(&data_path).await?;
217        let line = serde_json::to_string(&rec).unwrap();
218        f.write_all(line.as_bytes()).await?;
219        f.write_all(b"\n").await?;
220
221        if let Direction::Outbound = rec.direction {
222            if let Some(seq) = rec.seq {
223                let mut idx = OpenOptions::new().create(true).append(true).open(&idx_path).await?;
224                let idx_line = format!("{} {}\n", seq, offset);
225                idx.write_all(idx_line.as_bytes()).await?;
226            }
227        }
228
229        match cfg.durability {
230            DurabilityPolicy::Always => { let _ = f.sync_data().await; }
231            DurabilityPolicy::IntervalMs(ms) => {
232                if last_sync.elapsed() >= Duration::from_millis(ms) { let _ = f.sync_data().await; *last_sync = Instant::now(); }
233            }
234            DurabilityPolicy::Disabled => {}
235        }
236    }
237    Ok(())
238}

240#[async_trait]
241impl MessageStore for FileMessageStore {
242    async fn append(&self, record: StoredMessageRecord) -> std::io::Result<()> {
243        self.tx.send(record).await.map_err(|_| std::io::Error::new(std::io::ErrorKind::BrokenPipe, "storage channel closed"))
244    }
245
246    async fn append_bytes(&self, session: &SessionKey, direction: Direction, seq: Option<u32>, ts_millis: u64, payload: &[u8]) -> std::io::Result<()> {
247        let rec = StoredMessageRecord {
248            session: session.clone(),
249            direction,
250            seq,
251            ts_millis,
252            payload_b64: base64::encode(payload),
253        };
254        self.append(rec).await
255    }
256
257    async fn load_outbound_range(&self, session: &SessionKey, begin_seq: u32, end_seq: u32) -> std::io::Result<Vec<Bytes>> {
258        let stem = session.file_stem();
259        let data_path = self.cfg.base_dir.join(format!("{}.jsonl", stem));
260        let idx_path = self.cfg.base_dir.join(format!("{}.idx", stem));
261
262        // Read index and collect offsets
263        let idx_content = match fs::read_to_string(&idx_path).await { Ok(s) => s, Err(e) => {
264            if e.kind() == std::io::ErrorKind::NotFound { return Ok(Vec::new()); } else { return Err(e); }
265        }};
266        let mut offsets: Vec<(u32, u64)> = Vec::new();
267        for line in idx_content.lines() {
268            let mut it = line.split_whitespace();
269            let seq = it.next().and_then(|s| s.parse::<u32>().ok());
270            let off = it.next().and_then(|s| s.parse::<u64>().ok());
271            if let (Some(sq), Some(of)) = (seq, off) {
272                if sq >= begin_seq && sq <= end_seq { offsets.push((sq, of)); }
273            }
274        }
275        offsets.sort_by_key(|(s, _)| *s);
276
277        // Open data file once, then seek to read each record line
278        let mut file = File::open(&data_path).await?;
279        let mut out: Vec<Bytes> = Vec::with_capacity(offsets.len());
280        for (_seq, of) in offsets {
281            file.seek(std::io::SeekFrom::Start(of)).await?;
282            let mut reader = BufReader::new(&mut file);
283            let mut line = String::new();
284            reader.read_line(&mut line).await?;
285            if line.trim().is_empty() { continue; }
286            if let Ok(rec) = serde_json::from_str::<StoredMessageRecord>(&line) {
287                if let Ok(bytes) = base64::decode(&rec.payload_b64) {
288                    out.push(Bytes::from(bytes));
289                }
290            }
291        }
292        Ok(out)
293    }
294
295    async fn last_outbound_seq(&self, session: &SessionKey) -> std::io::Result<Option<u32>> {
296        let stem = session.file_stem();
297        let idx_path = self.cfg.base_dir.join(format!("{}.idx", stem));
298        let content = match fs::read_to_string(&idx_path).await { Ok(s) => s, Err(e) => {
299            if e.kind() == std::io::ErrorKind::NotFound { return Ok(None); } else { return Err(e); }
300        }};
301        let mut last: Option<u32> = None;
302        for line in content.lines() {
303            let seq = line.split_whitespace().next().and_then(|s| s.parse::<u32>().ok());
304            if let Some(sq) = seq { last = Some(last.map_or(sq, |m| m.max(sq))); }
305        }
306        Ok(last)
307    }
308}
